传入参数为函数并且该函数的参数也同时传入 




#include<iostream>

using namespace std;

template<class Fun,class... Args>

void runFunc(Fun &&f,Args&&... args){//注意空格啊

	f(args...);//注意内部怎么传参的啊

}

//总之注意以上的"..."

void f1(){

	cout << "f1" << endl;

}

void f2(int a){

	cout << "f2:" << a << endl;

}

void f3(int a,int b,int c){

	cout << "f3:" << a << b << c << endl;

}

void f4(int a,double d){

cout << "f4" << a << d << endl;

}



int main(){

	runFunc(f1);

	runFunc(f2,10);

	runFunc(f3,1,2,3);

        runFunc(f4,1,10.22);//可以是不同的类型

	return 0;

}
