	function array(obj){

		this.obj = obj;

		this.length = obj.length;

		this.getLength = function()

		{

			return this.length;

		}

		this.getClass = function(){

			return "array";

		}

		//return obj;//通过这一步实现了数组，但实际上此变成了函数

		//不能自定义实现数组

	}

	

	function $(obj){

		if(typeof obj == "string"){

			if(obj.startsWith("#") ){

				

				var element = document.getElementById(obj.replace("#",""));

				return element;

			}

			else if(obj.startsWith(".")){

				var element = document.getElementsByClassName(obj.replace(".",""));

				//alert(obj.replace(".",""));\\js中的正则表达式点不需要\\.

				return element;

			}

			else{

				try{

					var element = document.getElementsByTagName(obj);

					return element;

				}

				catch(e)

				{

					return null;

				}

			}

		}

		else if(typeof obj == "object"){

			if(obj.length == undefined){

				return obj;

			}

			else{

				return new array(obj);

			}

		}

		else if(typeof obj == "number"){

			return obj;

		}

		return null;

	}

	

	

	function f(){

		var ps1 = $("p");

		alert(num);

		alert(p.innerHTML);

		}

		catch(e)

		{

			alert(e);

		}

	}
