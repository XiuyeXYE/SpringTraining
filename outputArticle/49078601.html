<script>

	$(function(){

		$(":checkbox").change(function(){

			var $tr = $(":checked").parent().parent();

			$tr.css("backgroundColor","yellow");

			$tr = $(":not(:checked)").parent().parent();

			$tr.css("backgroundColor","white");

		});

		

	});

</script>
