#include<iostream>

using namespace std;



class Base{

	public:

	virtual void f(){

		cout << "Base::f" << endl;

	}

	virtual void g(){

		this->f();//this代表当前对象，在子类中代表子类对象

	}

};

class Derived:public Base{//默认为private继承

	public:

	void f(){

		cout << "Derived::f" << endl;

	}

};

int main(){

	Base b;

	Derived d;

	b.g();

	d.g();

	Base *p = new Derived;

	p->g();

	delete p;

	p = NULL;

	return 0;

}
