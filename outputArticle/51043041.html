<pre name="code" class="java">package com.test.collection;



import java.util.ArrayList;

import java.util.LinkedList;

import java.util.List;



public class ListTest {



	public static void main(String[] args) {

		

		List<Integer> array = new ArrayList<Integer>();

		List<Character> linked = new LinkedList<Character>();

		

		//普通的元素添加

		array.add(1);

		array.add(3);

		array.add(5);

		

		linked.add('A');

		linked.add('B');

		linked.add('C');

		

		System.out.println(array);

		System.out.println(linked);

		

		//在指定位置加入元素

		//void java.util.List.add(int index, Integer element)

		array.add(0,123);//index 从0开始

		

		linked.add(1,'X');

		

		System.out.println(array);

		System.out.println(linked);

		

		//验证是否包含数据

		if(array.contains(123)){

			System.out.println("Array contains 123");

		}

		else{

			System.out.println("Array not contains 123");

		}

		

		if(linked.contains('V')){

			System.out.println("Linked not contains V");

		}

		else{

			System.out.println("Linked not contains V");

		}

		

		//得到指定位置元素

		System.out.println(array.get(2));

		System.out.println(linked.get(2));

		

		//验证是否为空

		if(new ArrayList<String>().isEmpty()){

			System.out.println("ArrayList empty");

		}

		

		if(new LinkedList<Long>().isEmpty()){

			System.out.println("LinkedList empty");

		}

		

		if(!array.isEmpty() && !linked.isEmpty()){

			System.out.println("array and linked not empty");

		}

		

		//得到list的长度

		System.out.println(array.size());

		System.out.println(linked.size());

		

	} 

	



}
