windows 语音：根据文字说语音 


#define _USING_V110_SDK71_//vs 编译器一定要写这个宏，否则报错，还有就是sapi是c++的，只能用c++编译器编译

#include<sapi.h>

#include<stdio.h>

#include<sphelper.h>

int main(){

	

	::CoInitialize(NULL);

	ISpVoice *pSpVoice = NULL;

	if(FAILED(CoCreateInstance(CLSID_SpVoice,NULL,CLSCTX_INPROC_SERVER,IID_ISpVoice,(void**)&pSpVoice))){

		printf("Failed to create instance of ISpVoice!\n");

		return -1;

	}

	pSpVoice->Speak(L"Hello,Windows NT",SPF_DEFAULT,NULL);

	pSpVoice->Release();

	::CoUninitialize();

	printf("over");

	return 0;

}
