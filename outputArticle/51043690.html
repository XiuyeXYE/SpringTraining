package com.test.designpattern.proxy.silence;



public interface Server {

	void service();

}


package com.test.designpattern.proxy.silence;



public class RealServer implements Server{



	@Override

	public void service() {

		

		System.out.println("Hello Client,I am Server.I will service for you!");

		

	}



}




package com.test.designpattern.proxy.silence;



public class ProxyServer implements Server{



	private Server server = null;

	

	@Override

	public void service() {

		

		if(server == null){

			server = new RealServer();

		}

		

		System.out.println("Hello Client,I am Proxy Server,I will service for you.\nBut I don't really service for you.\nThe real server will service for you providing following something:");

		

		server.service();

	}

	

}




package com.test.designpattern.proxy.silence;



public class Main {



	public static void main(String[] args) {

		

		Server s = new ProxyServer();

		s.service();

		

	}

	

}

But I don't really service for you.

Hello Client,I am Server.I will service for you!




代理就是调用真正拥有者能力、作用、效果，或者在其原来的基础上增加一些新特性，相当于拥有者对代理授权使用，但其核心仍是拥有者的
