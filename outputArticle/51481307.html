package com.xiuye.son;



import java.util.ArrayList;

import java.util.List;



import com.xiuye.father.FatherSpi;



public class SonTOM implements FatherSpi {



	@Override

	public List<String> search() {

		List<String> list = new ArrayList<String>();

		list.add("Hello,");

		list.add("I'm TOM!");

		list.add("If you see me,I will talk to you!");

		list.add("Java Spi send me to here,");

		list.add("I don't know what happened!");

		list.add("Now , I'm here ,using spi is successful!AI...");

		return list;

		

	}



}
