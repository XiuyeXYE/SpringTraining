图片放大由主要由javascript实现 


<!doctype html>

<html>

<head>

<meta charset="utf-8">

<title>无标题文档</title>



<style type="text/css">



	.image{

		width:20%;

		height:20%;

	}

	

	/*

	仍然相当于放大为原图片大小的百分比！

	*/

	

	.big{

		width:50%;

		height:50%;

	}

	

</style>



<script>



	function showBig()

	{

		var img = document.getElementById("img");

		img.className="big";

		//主要为了过了一定时间后清除属性，还原

		var id = setTimeout(function(){

				img.className = "image";

				clearTimeout(id);

			},2000);

		

	}



</script>

</head>



<body>



<img id="img" class="image" src="../../images/1.jpg" onMouseOver="showBig();"/>



</body>

</html>
