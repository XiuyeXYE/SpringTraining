package com.xiuye.algorithm;



public class Perm {



	//全排序，k指定开始的下标，m指定数组长度，list要全排序的数组

	public static void perm(int list[],int k,int m)

	{

		if(k == m)

		{

			for(int i=0;i<m;i++)

			{

				System.out.print(list[i]);

			}

			System.out.println();

		}

		else

			for(int i = k;i<m ;i++)

			{

				int tmp = list[k];

				list[k] = list[i];

				list[i] = tmp;

				perm(list,k+1,m);

				tmp = list[k];

				list[k] = list[i];

				list[i] = tmp;

				

			}

	}

	

	public static void perm(int list[],int k)

	{

		if(k == list.length)

		{

			for(int i=0;i<list.length;i++)

			{

				System.out.print(list[i]);

			}

			System.out.println();

		}

		else

			for(int i = k;i<list.length ;i++)

			{

				int tmp = list[k];

				list[k] = list[i];

				list[i] = tmp;

				perm(list,k+1);

				tmp = list[k];

				list[k] = list[i];

				list[i] = tmp;

				

			}

	}

	

	

	public static<T>void swap(T m,T n)//失败交换函数

	{

		T tmp = m;

		m = n;

		n = tmp;

		//System.out.println(m+":"+n);

	}

	//引用本质是值传递，传递实参，要拷贝副本，即形参，交换的栈空间地址

	//没有对堆内数据进行更改所以一定要注意。仅仅传的是拷贝的副本引用。

	public static void swap(MyInteger a,MyInteger b)

	{

		int tmp = a.getValue();

		a.setValue(b.getValue());

		b.setValue(tmp);

	}
