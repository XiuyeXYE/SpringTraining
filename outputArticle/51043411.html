package com.test.io;



import java.io.File;



public class FileTest {



	public static void main(String[] args) {



		/**

		 * 不管文件存不存在File都可得到实例对象，不会报异常（NullPointerException）

		 * eclipse 下的src下的资源文件，导出工程所在目录的bin目录下，

		 * 

		 */

		File f = new File("bin", "test.txt");

		if (f != null) {

			System.out.println(f);

			System.out.println("Parent Path:"+f.getParent());

			System.out.println("Relative Path:" + f.getPath());//相对位置

			System.out.println("Absolute Path:" + f.getAbsolutePath());

			if(f.exists()){

				System.out.println("test.txt exits");

				System.out.println(f.getName());

				System.out.println("是否是文件:" + f.isFile());

				System.out.println("是否是目录:" + f.isDirectory());

				

			}

			else{

				System.out.println("test.txt not exits");

			}

			

		}

		

		File f1 = new File("bin","notExist.txt");

		

		if(f1 != null){

			System.out.println(f1);

			System.out.println("Parent Path:"+f1.getParent());

			System.out.println("Relative Path:" + f1.getPath());

			System.out.println("Absolute Path:" + f1.getAbsolutePath());

			if(f1.exists()){

				System.out.println("notExist.txt exits");

				

			}

			else{

				System.out.println("notExist.txt not exits");

				System.out.println(f1.getName());

				System.out.println("是否是文件:" + f1.isFile());

				System.out.println("是否是目录:" + f1.isDirectory());

				

			}

		}

		/**

		 * 直接文件名表示当前目录

		 * 

		 */

		File f2 = new File("test.txt");

		if(f2 != null){

			System.out.println(f2);

			System.out.println(f2.getAbsolutePath());	

			if(f2.exists()){

				System.out.println("test.txt exits");

			}

			else{

				System.out.println("test.txt not exits");

			}

		}

		

		

		File f3 = new File("notExist.txt");

		if(f3 != null){

			System.out.println(f3);

			System.out.println(f3.getAbsolutePath());	

			if(f3.exists()){

				System.out.println("notExist.txt exits");

			}

			else{

				System.out.println("notExist.txt not exits");

			}

		}

	}



}
