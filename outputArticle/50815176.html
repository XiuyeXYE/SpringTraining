直接给字符指针赋字符串 


#include<iostream>

#include<string>

int main(){

	using namespace std;

	string s;

	cin >> s;

	cout << sizeof(s.c_str()) << endl;

	cout << sizeof(s) << endl;

	cout << s << endl;

	char * cs = new char[10];



	cin >> cs ;

	cout << sizeof(cs) << endl;

	cout << cs << endl;

	delete []cs;

	return 0;

	

}
如果cs未分配空间在执行cin后就会直接运行异常直接终止。所以要先给字符指针分配空间后赋值，运行正常。
