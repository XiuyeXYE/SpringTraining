Objective-C Rectangle Demo 


#import <Foundation/Foundation.h>



@interface Rectangle:NSObject

{

    int width;

    int height;

}



-(void)setWidth:(int)_w;

-(void)setHeight:(int)_h;

-(void)setWH:(int)_w:(int)_h;

-(int)getArea;

-(int)getCircumference;



@end


#import "Rectangle.h"



@implementation Rectangle



-(void)setWidth:(int)_w{

    width = _w;

}

-(void)setHeight:(int)_h{

    height = _h;

}

-(void)setWH:(int)_w:(int)_h{

    width = _w;

    height = _h;

}

-(int)getArea{

    return width * height;

}

-(int)getCircumference{

    return 2 * (width + height);

}



@end;


#import <Foundation/Foundation.h>

#import "header/Rectangle.h"



int main (int argc, const char * argv[])

{

   NSAutoreleasePool * pool = [[NSAutoreleasePool alloc] init];



   NSLog (@"hello world");

   Rectangle *rect = [[Rectangle alloc] init];

  

   [rect setWidth:123];

   [rect setHeight:456];

   int area = [rect getArea];

   int length = [rect getCircumference];

   NSLog(@"Area = %d",area);

   NSLog(@"Circumference = %d",length);

   

   [rect setWH:50:50];

   area = [rect getArea];

   length = [rect getCircumference];

   NSLog(@"Area = %d",area);

   NSLog(@"Circumference = %d",length);

   

   [pool drain];

   return 0;

}
