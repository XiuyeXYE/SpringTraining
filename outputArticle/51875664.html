OpenMP First Demo 


#include<iostream>

using namespace std;

int main(){

	

	#pragma omp parallel

	{

		cout << "All Threads run this" << endl;

		#pragma omp sections

		{

			#pragma omp section

			{

				cout << "This executes in parallel" << endl;

			}

			#pragma omp section

			{

				cout << "Sequential statement 1" << endl;

				cout << "This always executes after statement 1" << endl;

			}

			#pragma omp section

			{

				cout << "This also executes in parallel" << endl;

			}

		}

	}



}
