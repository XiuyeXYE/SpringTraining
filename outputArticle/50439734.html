select  a.name+' ' +case a.usertype when 7 then 'int' 

		when 6 then 'smallint'

		when 2 then 'varchar'

		when 11 then 'money'

		when 1 then 'char'

		when 8 then 'float'

		when 12 then 'datetime'

		when 19 then 'text'

		else 'other' end

	      +case when a.usertype in(2,1,8 ) then '('+convert(varchar(4),a.length)+')' end + " "

		+case when a.status=0 then 'not NULL' else 'NULL' end + ","

from syscolumns a join sysobjects b on a.id = b.id

where b.name = 'order_header'

order by a.colid

go
