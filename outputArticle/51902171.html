#include<iostream>

using namespace std;

int main(){





#ifdef __cplusplus

        cout << "__cplusplus:" << __cplusplus << endl;

#endif





#ifdef _WIN32

        cout << "_WIN32:"<< _WIN32 <<endl;

#endif

#ifdef __APPLE__

        cout << "__APPLE:" << __APPLE__ << endl;

#endif

#ifdef __linux__

        cout << "__linux__:" << __linux__ << endl;

#endif

#ifdef _MSC_VER

        cout << "_MSC_VER:" << _MSC_VER << endl;

#endif

#ifdef __INTEL_COMPLIER

        cout << "__INTEL_COMPLIER:" << __INTEL_COMPILER << endl;

#endif

#ifdef __clang__

        cout << "__clang__:" << __clang__ << endl;

#endif

#ifdef __GNUC__

        cout << "__GNUC__:" << __GNUC__ << endl;

#endif

#ifdef __GNUC_MINOR__

        cout << "__GNUC_MINOR__:" << __GNUC_MINOR__ << endl;

#endif

#ifdef __GNUC_PATCHLEVEL__

        cout << "__GNUC_PATCHLEVEL__:" << __GNUC_PATCHLEVEL__ << endl;

#endif









        return 0;

}
