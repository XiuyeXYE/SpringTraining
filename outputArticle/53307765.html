spring mvc ResponseBody 输出乱码解决方案 


@RequestMapping(value = "article.sb", produces = "text/html; charset=utf-8")

	@ResponseBody

	public String article(int type, String id) {



		String content = null;



		switch (type) {

		case 0:

		case 4:

		case 5:

			content = this.articleDao.findByIdAndType(id, type);

			break;



		}

		log.debug("content := " + content);



		return content;



	}
