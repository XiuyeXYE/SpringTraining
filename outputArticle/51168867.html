结构体指定成员赋值 


#include<iostream>

using namespace std;

void f(){

	cout << "Hello" << endl;

}



int main(){

	struct Test{

		void (*function)();

		string name;

	};

	Test t = {.function=f,.name="World"};

	t.function();

	cout << t.name << endl;

	return 0;



}
