using namespace std;



template<typename T>

class Object

{

	public:

		void f(T t = 0){

			cout << t << endl;

		}

		virtual ~Object(){

			cout << "Object released" << endl;

		}

	



};



class Sub:public Object<int>

{

	public:

		~Sub(){

		cout << "Sub released" << endl;

	}

	

};



class SubSub:public Sub

{

	public:

		~SubSub(){

		cout << "SubSub released" << endl;

	}

};



template<>

class Object<double>

{

	

	public:

		void f(double t){

			cout << t << endl;

	}

	

};
