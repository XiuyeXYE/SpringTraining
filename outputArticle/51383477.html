Template's Template 


#include <cstdio>

#include <tuple>



template< class, class > struct S {

  S() { puts("primary"); }

};



template<

  template< class... > class T, class...TArgs

, template< class... > class U, class...UArgs

>

struct S< T< TArgs... >, U< UArgs... > > {

  S() { puts("specialized"); }

};



int main()

{

  S< int, int >  p;                                       // "primary"

  S< std::tuple< int, char >, std::tuple< float > >  s;   // "specialised"

}
